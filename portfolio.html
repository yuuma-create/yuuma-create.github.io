レシピの賞味期限表示ロジック（utils.dart）

class ExpiryInfo {
  final int displayDays;
  final Color expiryColor;
  final String expiryText;
  ExpiryInfo(this.displayDays, this.expiryColor, this.expiryText);
}

ExpiryInfo getExpiryInfo(int? daysUntilExpiry) {
  int displayDays = daysUntilExpiry ?? 999;
  Color expiryColor = Colors.grey;
  String expiryText = "";
  if (displayDays < 0) {
    expiryColor = Colors.red.shade700;
    expiryText = "(期限切れ)";
  } else if (displayDays == 0) {
    expiryColor = Colors.red;
    expiryText = "(あと0日)";
  } else if (displayDays <= 3) {
    expiryColor = Colors.red;
    expiryText = "(あと $displayDays 日)";
  } else if (daysUntilExpiry != null) {
    expiryColor = themeColorNotifier.value;
    expiryText = "(あと $displayDays 日)";
  }
  return ExpiryInfo(displayDays, expiryColor, expiryText);
}
